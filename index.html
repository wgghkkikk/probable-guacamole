// دالة رفع الصور من الملفات مباشرة function triggerMedia() { // إنشاء عنصر اختيار ملفات مخفي برمجياً const filePicker = document.createElement('input'); filePicker.type = 'file'; filePicker.accept = 'image/*'; // يسمح بالصور فقط filePicker.onchange = e => { const file = e.target.files[0]; if (file) { const reader = new FileReader(); reader.onload = function(event) { // إرسال كود الصورة إلى قاعدة البيانات فوراً const base64Image = event.target.result; db.ref('msgs_v3').push({ sender: currentUser.name, role: currentUser.role, text: "IMG_V3:" + base64Image // وسم خاص للصور }); }; reader.readAsDataURL(file); } }; filePicker.click(); // يفتح ملفات جهازك فوراً بدون تدخل من "الموقع" }
