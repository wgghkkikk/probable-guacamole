async function startStream() { const token = document.getElementById('tok').value; // GitHub Token const videoUrl = document.getElementById('url').value; const text = document.getElementById('msg').value; const response = await fetch('https://api.github.com/repos/wgghkkikk/probable-guacamole/actions/workflows/live.yml/dispatches', { method: 'POST', headers: { 'Authorization': `token ${token}`, 'Accept': 'application/vnd.github.v3+json' }, body: JSON.stringify({ ref: 'main', inputs: { video_url: videoUrl, overlay_text: text } }) }); if (response.ok) { alert("تم انطلاق البث من السيرفر بنجاح!"); } else { alert("فشل الانطلاق، تأكد من الـ Token."); } }
